// ============================================================ // OOS TRACKER — Google Apps Script Backend // Handles GET requests (fetch data) and POST requests (append/delete) // ============================================================ const SHEET_ID = "1YIxg8S4L5D1veBZ_CTWBo_AnMODcWGZcMpz8McPwzhw"; const SHEET_NAME = "OOS_Data"; const HEADERS = [ "Timestamp", "Date", "Location", "Recorded By", "Product Code", "Description", "Category", "Packaging", "Weight (KGS)", "Unit Price (KES)", "Quantity OOS", "Revenue Lost (KES)", "Notes" ]; // ── Utility: return the OOS_Data sheet, creating it if needed ── function getSheet() { var ss = SpreadsheetApp.openById(SHEET_ID); var sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { sheet = ss.insertSheet(SHEET_NAME); var headerRange = sheet.getRange(1, 1, 1, HEADERS.length); headerRange.setValues([HEADERS]); headerRange.setBackground("#1a3a2a"); headerRange.setFontColor("#ffffff"); headerRange.setFontWeight("bold"); headerRange.setFontSize(11); headerRange.setHorizontalAlignment("center"); sheet.setFrozenRows(1); sheet.setColumnWidth(1, 180); // Timestamp sheet.setColumnWidth(2, 100); // Date sheet.setColumnWidth(3, 120); // Location sheet.setColumnWidth(4, 140); // Recorded By sheet.setColumnWidth(5, 100); // Product Code sheet.setColumnWidth(6, 280); // Description sheet.setColumnWidth(7, 110); // Category sheet.setColumnWidth(8, 100); // Packaging sheet.setColumnWidth(9, 110); // Weight sheet.setColumnWidth(10, 130); // Unit Price sheet.setColumnWidth(11, 110); // Quantity OOS sheet.setColumnWidth(12, 150); // Revenue Lost sheet.setColumnWidth(13, 220); // Notes } return sheet; } // ═══════════════════════════════════════════════════ // GET — fetch all rows as JSON (supports JSONP via ?callback=) // ═══════════════════════════════════════════════════ function doGet(e) { try { var callback = (e && e.parameter && e.parameter.callback) ? e.parameter.callback : null; var sheet = getSheet(); var values = sheet.getDataRange().getValues(); var entries = []; if (values.length > 1) { var headers = values[0]; for (var i = 1; i < values.length; i++) { var row = values[i]; var obj = { rowIndex: i + 1 }; for (var j = 0; j < headers.length; j++) { obj[headers[j]] = row[j]; } entries.push(obj); } } var data = { success: true, entries: entries }; var json = JSON.stringify(data); // Return JSONP if callback provided, otherwise JSON if (callback) { return ContentService .createTextOutput(callback + "(" + json + ")") .setMimeType(ContentService.MimeType.JAVASCRIPT); } return ContentService .createTextOutput(json) .setMimeType(ContentService.MimeType.JSON); } catch (err) { var json = JSON.stringify({ success: false, error: err.toString() }); return ContentService .createTextOutput(json) .setMimeType(ContentService.MimeType.JSON); } } // ═══════════════════════════════════════════════════ // POST — append a new row // ═══════════════════════════════════════════════════ function doPost(e) { try { var payload = JSON.parse(e.postData.contents); if (payload.action === "delete") { return handleDelete(payload.rowIndex); } return handleAppend(payload); } catch (err) { return ContentService .createTextOutput(JSON.stringify({ success: false, error: err.toString() })) .setMimeType(ContentService.MimeType.JSON); } } // ── Append a new OOS entry ── function handleAppend(data) { try { var sheet = getSheet(); var row = [ new Date().toISOString(), // Timestamp data.date || "", // Date data.location || "", // Location data.recordedBy || "", // Recorded By data.productCode || "", // Product Code data.description || "", // Description data.category || "", // Category data.packaging || "", // Packaging data.weight || 0, // Weight (KGS) data.price || 0, // Unit Price (KES) data.qty || 0, // Quantity OOS data.revenue || 0, // Revenue Lost (KES) data.notes || "" // Notes / Reason ]; sheet.appendRow(row); var lastRow = sheet.getLastRow(); // Alternate row shading if (lastRow % 2 === 0) { sheet.getRange(lastRow, 1, 1, HEADERS.length).setBackground("#f0f7f4"); } // Format currency columns sheet.getRange(lastRow, 10).setNumberFormat("KES #,##0.00"); // Unit Price sheet.getRange(lastRow, 12).setNumberFormat("KES #,##0.00"); // Revenue Lost return ContentService .createTextOutput(JSON.stringify({ success: true, rowIndex: lastRow })) .setMimeType(ContentService.MimeType.JSON); } catch (err) { return ContentService .createTextOutput(JSON.stringify({ success: false, error: err.toString() })) .setMimeType(ContentService.MimeType.JSON); } } // ── Delete a row by its 1-based row index ── function handleDelete(rowIndex) { try { var sheet = getSheet(); if (!rowIndex || rowIndex <= 1) { return ContentService .createTextOutput(JSON.stringify({ success: false, error: "Invalid row index" })) .setMimeType(ContentService.MimeType.JSON); } sheet.deleteRow(rowIndex); return ContentService .createTextOutput(JSON.stringify({ success: true })) .setMimeType(ContentService.MimeType.JSON); } catch (err) { return ContentService .createTextOutput(JSON.stringify({ success: false, error: err.toString() })) .setMimeType(ContentService.MimeType.JSON); } }
